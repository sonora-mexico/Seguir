(function() {
  const found = new Set();
  const skip = new Set(['home','explore','notifications','messages','settings','i','search','compose','intent','login','signup']);
  document.querySelectorAll('[data-testid="UserCell"]').forEach(cell => {
    cell.querySelectorAll('a[href]').forEach(a => {
      const m = (a.getAttribute('href')||'').match(/^\/([a-zA-Z0-9_]{1,50})(\?.*)?$/);
      if (m && m[1] && !skip.has(m[1].toLowerCase())) found.add('@' + m[1].toLowerCase());
    });
  });
  const out = [...found].join('\n');
  console.log('%c=== X Follower Analyzer by @MexiSonora ===','color:#1d9bf0;font-weight:bold;font-size:13px');
  console.log('%c' + found.size + ' handles encontrados','color:#00c47a;font-weight:bold');
  console.log(out);
  try { copy(out); console.log('%c✓ Copiado al portapapeles','color:#f5c000'); } catch(e){}
})()
